{

	q ..RecordDataNew(Index,Ginseng,Data,Type,DayNumber)


	s Ginseng=$g(Ginseng),Data=$g(Data),DayNumber=$g(DayNumber),Index=$g(Index),Type=$g(Type),OrderNum=$g(OrderNum)

MIFFUS3000HL7(mi)
	s mi=$g(mi) 
	S ^TMPMACH10(mi,104)=$H
	i '$l(mi) q
	s ItemDeli=$li(^dbo.BTMIMachineParameterD(mi),12) //项目分隔符
	s ResultDeli=$li(^dbo.BTMIMachineParameterD(mi),13) //结果分隔符
	s par10=ResultDeli	//结果分隔符
	s par11=ItemDeli	//项目分隔符
	s AntDeli=$li(^dbo.BTMIMachineParameterD(mi),14) //抗生素分隔符
	s SenDeli=$li(^dbo.BTMIMachineParameterD(mi),15) //药敏结果分隔符
	s Port="|TCP|"_$li(^dbo.BTMIMachineParameterD(mi),10)  //端口号
	//控制字符
  	s stx=$c(2),etx=$c(3),ack=$c(6),enq=$c(5),eot=$c(4),etb=$c(23)
 	s lf=$c(10),cr=$c(13),nak=$c(21),(AllRecord,result,epis)="",kk=1
 	s SB=$C(11),EB=$C(28),CR=$c(13)
 	s MSH="MSH|^~\&|LIS||FUS100||20090424130516||ACK|ACK0000001|P|2.3"
	S ^TMPMACH10(mi,102)=$H
	Q:$$Start^MI.MIF000(mi)
	s par4="|TCP|"_$li(^dbo.BTMIMachineParameterD(mi),10)  	//端口号
	s IP=$li(^dbo.BTMIMachineParameterD(mi),9)
	s Port=$li(^dbo.BTMIMachineParameterD(mi),10)
	s LinkTime=0
	s Device="|TCP|"_Port
	F {
		try{
			;d Trace^MI.MIF000(mi,$ZA,"ZA")
			//s ^LISMachineLog("YHL3000","$ZA",$zd($p($h,",",1),8),$p($zt($p($h,",",2)),":",1),$zt($p($h,",",2)))=$ZA
			//未启动客户端/服务端0，客户端断开连接标志20484，无客户端连接20480，有客户端连接28674，连接并建立数据传输28672
			//lis断开
			i $ZA="0" {
				h 2
				open Device:(IP:Port:/IOTABLE="GB18030"):20
				use Device
				d Trace^MI.MIF000(mi,"lis断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"lisReConnect")
				//throw ##class(%Exception.General).%New("ConnectError","D",,"重新连接失败")
				}
			//仪器断开，或者无数据1min，重启tcp
			;i (($ZA="20484")||(LinkTime=6)) {
			i (($ZA="20484")) {
				close par4
				d Trace^MI.MIF000(mi,"仪器断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"blockConnect")
				h 2
				open Device:(IP:Port:/IOTABLE="GB18030"):20
				use Device
				d Trace^MI.MIF000(mi,"重新连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"reConnect")
				s LinkTime=0
				continue
				}
			//连接无数据传输，main每执行6次，即1min，重启tcp
			i $ZA="28674" s LinkTime=$i(LinkTime)
			//正常传输了，就清空计数
			i $ZA="28672" s LinkTime=0
			d Main
	 	}catch ex{
		 s exception="错误:"_ex.Location_"^"_ex.Data_ex.Name
		 d Trace^MI.MIF000(mi,exception,"读取或写入异常，重启服务端")
	 	}
		Q:$$Stop^MI.MIF000(mi)
	}
    C par4
    Q
	/*
	i $$Start^MI.MIF000(mi) q
	f  d Main i $$Stop^MI.MIF000(mi) q
  	c Port
  	q
  	*/
  
ErrHandler
    h 10
    s iError=+$g(iError)+1
    i iError>100 s ret=$$Stop^MI.MIF000(mi)  q
    d Trace^MI.MIF000(mi,$tr($ZERROR,"^","--")_".错误代码:"_$ECODE,"Runtime Error")
    q
  
  
Main   
	SET $ZTRAP="ErrHandler" S iERR=0
	s Allrecord=$$Read^MI.MIF000(mi,SB,EB,"","",2) i '$l(Allrecord) q
	s (sample,epis,surname,result,date,time,QC)=""

	for i=1:1:$l(Allrecord,cr) {
		s record=$p(Allrecord,cr,i)
		q:'$l(record)
		D Trace^MI.MIF000(mi,record,"H<--M")
		S segmentType=$P(record,"|",1)
		;消息段(数据类型) 	MSH	  MSA 	PID   OBR  	OBX  ORD  QRF 	ERR   QAK    DSP     DSC   
		I segmentType="MSH" {   //信息头信息
		    S mshStr=record
		    //消息类型
		    S msgType=$P($P(record,"|",9),"^",1)  //ORU,ACK,QRY,QCK,DSR 
		    I msgType="ORU" {
		        D SendACK(mi,record,"AA",0,"Message accepted")
		    }
		    I msgType="ORM" {
		    }
		    I msgType="QRY" {
		        D SendQryACK(mi,record,"AA",0,"Message accepted")
		    }
		    S resultType=$P(record,"|",11)
		}

		I segmentType="PID" {   //患者信息
		      
		}

		I segmentType="PVI" {   //就诊信息
		      
		}

		I segmentType="OBR" {   //医嘱信息
		    S epis=$P(record,"|",3) 
		    S:'$l(epis) epis=$P(record,"|",4)
		    S dateStr=$P(record,"|",8)
		    I dateStr?14N {
		        S date=$ZDH($E(dateStr,1,4)_"-"_$E(dateStr,5,6)_"-"_$E(dateStr,7,8),3)
		        S time=$ZTH($E(dateStr,9,10)_":"_$E(dateStr,11,12)_":"_$E(dateStr,13,14),1) 
		    }
		    S code=$P($P(record,"|",5),"^",1)
		}

		I segmentType="OBX" {
		    S dataTye=$P(record,"|",3)  //NM (numeric) 表示数字值，用于定量项目,
		                                //ST (string) 表示字符串值，用于定性项目
		    S channel=$P(record,"|",4)
		    S channel=$P(channel,"^",1)
		    S value=$P(record,"|",6)
		  
		    S result=result_channel_ResultDeli_value_ItemDeli
		}

		//根据条码号获取LIS标本
		I segmentType="QRD" {
		    S barcode=$P(record,"|",9)  //QRD|20161208150935|R|D|1|||RD|16120703726|OTH|||T|
		    I $L(barcode) {
		        S ^TMPMACHINEEPIS(mi,"B",barcode,"MSH")=mshStr
		        S ^TMPMACHINEEPIS(mi,"B",barcode,"QRD")=record
		    }
		}
		//根据查询条件获取LIS标本
		I segmentType="QRF" { 
			S tRecord=record
			S record=$TR(record," ")
			S qryStartDate=$P(record,"|",3)
		    S qryEndDate=$P(record,"|",4)
		    S sampleIdStart=$P(record,"|",5)
		    S sampleIdEnd=$P(record,"|",6)
		    S:'$L(qryStartDate) qryStartDate=qryEndDate
		    S:'$L(qryEndDate) qryEndDate=qryStartDate
		    S:'$L(sampleIdStart) sampleIdStart=sampleIdEnd
		    S:'$L(sampleIdEnd) sampleIdEnd=sampleIdStart
		    //通过样本号段查询  QRF|iFlash3000|||1|9|RCT|COR|ALL|| 
		    I sampleIdStart?1.N,sampleIdEnd?1.N {
		        F sampleId=sampleIdStart:1:sampleIdEnd {
		            S ^TMPMACHINEEPIS(mi,"S",sampleId,"MSH")=mshStr
		            S ^TMPMACHINEEPIS(mi,"S",sampleId,"QRF")=tRecord
		        }
		    }
		    //通过时间段查询	QRF|iFlash3000|20160122080000|20160122120000|||RCT|COR|ALL||
		    I qryStartDate?1.N,qryEndDate?1.N {
		        F qryDate=qryStartDate:1:qryEndDate {
		            S ^TMPMACHINEEPIS(mi,"D",qryDate,"MSH")=mshStr
		            S ^TMPMACHINEEPIS(mi,"D",qryDate,"QRF")=tRecord
		        }
		    }
		}
		
	}

	D Save

	D Order

	q

Save

    S barcode="",barcodeIndex=0

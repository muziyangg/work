MIFEXC810(mi) 
	///仪器名称:	全自动血凝分析仪
	///仪器型号:	迈瑞EXC810
	///仪器厂商:	迈瑞
	///通信协议:	HL7
	///交互方式:	双 向
	///编 写 者:	LCH
	///编写日期:	2019-09-11
	///接    口:	TCP/IP,RS232
	///备    注:	仪器配置:选择HL7协议,散点图选择:位图   
    S mi=$G(mi) 
    Q:'$L(mi)
  
	s par11=$li(^dbo.BTMIMachineParameterD(mi),12) 		//项目分隔符
	s par10=$li(^dbo.BTMIMachineParameterD(mi),13) 	//结果分隔符
	s AntDeli=$li(^dbo.BTMIMachineParameterD(mi),14) 		//抗生素分隔符
	s SenDeli=$li(^dbo.BTMIMachineParameterD(mi),15) 		//药敏结果分隔符
	s par4="|TCP|"_$li(^dbo.BTMIMachineParameterD(mi),17)  	//端口号
    S stx=$C(2),etx=$C(3),ack=$C(6),enq=$C(5),eot=$C(4),etb=$C(23)
    S sb=$C(11),eb=$C(28),cr=$C(13)  
    S lf=$C(10),nak=$C(21),(AllRecord,result,epis)="",kk=1
    S $ZT="ErrHandler",$EC=""
    S (sample,epis,surname,rec,res,result,date,time,QC,errFlag,resultType)=""
    Q:$$StartUTF8^MI.MIF000(mi)
	F {
		D Main
		Q:$$Stop^MI.MIF000(mi)
	}
    C par4
    Q
  
Main R *R:10
	Q:'$TEST	//如果超时则退出
	Q:$c(R)'=sb   //以sb开头
	S (sample,epis,surname,rec,res,result,date,time,QC,errFlag,resultType,msgType,barcode,mshStr)=""
	S startTime=$P($H,",",2)
	K PLIST
	S index=1
	F {
		S curTime=$P($H,",",2)
		Q:curTime>(startTime+300)  //5分钟内没有处理完则退出循环
		Q:$$Stop^MI.MIF000(mi)
		R *R:10
		Q:'$TEST	//如果超时则退出
		I $L($G(PLIST(index)))>=32700 {  //32767
			S index=+$O(PLIST(""),-1)+1
			S PLIST(index)=""
		}
		I $C(R)'=eb,$C(R)'=cr {
			S PLIST(index)=$G(PLIST(index))_$C(R)
		}
		I $C(R)=cr {  	//解析一条记录
			D Record
			K PLIST
			S index=1
		}
		Q:$C(R)=eb   	//循环读取,当为eb的时候则退出
	}
	D Trace^MI.MIF000(mi,result,"result")
	D Save
	Q

Record
    S index=""
    S segmentType="",dataTye="",graphType="",base64Stream=""
    S isImage=0,isBinary=0
    F {
		S index=$O(PLIST(index))
		Q:'$L(index)
		S record=$TR($G(PLIST(index)),cr)
		Continue:'$L(record)
		D Trace^MI.MIF000(mi,record,"H<--M")
		I index=1 {
			S segmentType=$P(record,"|",1)
			I segmentType="MSH" {	//信息头信息
			   //MSH|^~\&|||||20180814165535||QRY^Q02|2|P|2.3.1||||||ASCII|
				S mshStr=record
				//消息类型
    			S msgType=$P($P(record,"|",9),"^",1)  //ORU,ACK,QRY,QCK,DSR
    			I msgType="ORU" {
	    			D SendACK(mi,record,"AA",0,"Message accepted")
	    		}
	    		//I msgType="QRY" {
	    		///	D SendQACK(mi,record,"AA",0,"Message accepted")
	    		//}
	    		I msgType="ORM" {
	    		}
	    		S resultType=$P(record,"|",11)
			}
		
			I segmentType="PID" {	//患者信息
				
			}
		
			I segmentType="PVI" {	//就诊信息
				
			}
		
			I segmentType="OBR" {	//医嘱信息
				S epis=$P(record,"|",3) 
    			S:'$l(epis) epis=$P(record,"|",4)
	     		S code=$P($P(record,"|",5),"^",1)
	     		I code="00003" {	//LJ QCR
		     	
		     	}
			}
		
		
			I segmentType="OBX" {
				S dataTye=$P(record,"|",3)  //NM (numeric) 表示数字值，用于定量项目,
     										//ST (string) 表示字符串值，用于定性项目
    			S channel=$P(record,"|",4)
    			S channel=$P(channel,"^",2)
    			S value=$P(record,"|",6)
    		
    			I channel="SI" {	//value=浊度(L)^溶血(H)^黄疸(I)
	    			S:$l($P(value,"^",2)) result=result_channel_"H"_par10_$P(value,"^",2)_par11
	    			S:$l($P(value,"^",3)) result=result_channel_"I"_par10_$P(value,"^",3)_par11
	    			S value=$P(value,"^",1)
	    		}
    			I dataTye="ST" {
    				S value=$S(value="ò?D?":"阴性",value="??D?":"阳性",1:value)
    			}
    			I $L(channel),dataTye'="ED",dataTye'="IS" {
	    			S result=result_channel_par10_value_par11
    			}
    		
    			I dataTye="IS" {	  
	    			I channel="05001",resultType="Q" {		//质控浓度
		    			S epis=value
		    		}
	    		}
    			S graphTypeDesc=$ZCVT($P($P(record,"|",4),"^",2),"U")
    		
    			I dataTye="ED" {	//图形数据
    				I $ZCVT($P(value,"^",2),"U")=$ZCVT("Image","U") {
	    				S isImage=1  	//位图数据
	    			}
	    			I $ZCVT($P(value,"^",3),"U")=$ZCVT("Octer-stream","U") {
		    			S isBinary=1	//二进制数据(需要绘图)
		    		}
    				S graphType=channel
    				I channel="15050" {	//二进制数据
	    				S graphType="RBC"
	    			}
	    			I channel="15056" {	//位图数据
	    				S graphType="RBC"
	    			}
	    			I channel="15100" {	//二进制数据
	    				S graphType="PLT"
	    			}
	    			I channel="15116" {	//位图数据
	    				S graphType="PLT"
	    			}
	    			I channel="15200" {	//位图数据
	    				S graphType="WBC-DIFF"
	    			}
	    			I channel="15201" {	//二进制数据
	    				S (isImage,isBinary)=0	//暂时不处理二进制数据
	    				S graphType="WBC-DIFF"
	    			}
					I channel="15202" {	//二进制数据
	    				S (isImage,isBinary)=0	//暂时不处理二进制数据
	    				S graphType="WBC-DIFF"
	    			}
	    			I channel="15250" {	//位图数据
	    				S graphType="BASO"
	    			}
	    			I channel="15251" {	//二进制数据
	    				S (isImage,isBinary)=0	//暂时不处理二进制数据
	    				S graphType="BASO"
	    			}
	    			I channel="15300" {	//位图数据
	    				S graphType="RET"
	    			}
	    			I channel="15306" {	//二进制数据
	    				S (isImage,isBinary)=0	//暂时不处理二进制数据
	    				S graphType="RET"
	    			}
	    			I channel="15301" {	//位图数据
	    				S graphType="PLT-0"
	    			}
	    			I channel="15302" {	//位图数据
	    				S graphType="RET-EXT"
	    			}
	    			I channel="15350" {	//位图数据
	    				S graphType="NRBC"
	    			}
	    			I channel="15354" {	//二进制数据
	    				S (isImage,isBinary)=0	//暂时不处理二进制数据
	    				S graphType="NRBC"
	    			}
	    			S graphData=$P(value,"^",5)
	    		
	    			S graphDataType=$P(value,"^",4)
	    			I $ZCVT(graphDataType,"U")=$ZCVT("Base64","U") {
	    				S base64Stream=##Class(%GlobalCharacterStream).%New()
	    				D base64Stream.Write(graphData)
	    			}
	    		}
			}
		
			I segmentType="ORC" {	//就诊信息
				S barcode=$P(record,"|",4)
				//S:$ZCVT(barcode,"U")=$ZCVT("Invalid","U") barcode=""  //条码扫描错误
				//I $L(barcode) {  //双向
				//} Invalid 条码扫描错误
				d SendORR(mi,record,"AA",0,"Message accepted",mshStr,barcode)
			}
			I segmentType="QRD" {	//就诊信息
				S barcode=$P(record,"|",9)
				//d SendDSR(mi,record,"AA",0,"Message accepted",mshStr,barcode)
			}
			I segmentType="QRF" {	//就诊信息
			    h 0.5
				D SendQACK(mi,mshStr,"AA",0,"Message accepted")
				h 0.5
				d SendDSR(mi,record,"AA",0,"Message accepted",mshStr,barcode)
			}
		}
	
		I dataTye="ED",index>1,$L(base64Stream) {
			D base64Stream.Write($P(record,"|",1))
		}
	}
  
	I $L(base64Stream) {  //保存图形
		S graphOrder=0
		I isImage {   //位图格式
			S title=graphType
			S error=##Class(MI.Instrument).SaveBase64StreamToGraph(base64Stream,mi,epis,graphType,graphOrder,title,"")
			I $L(error) {
				D Trace^MI.MIF000(mi,$TR(error,"^","--"),"Graph Error")
			}
		} ELSE {
			I isBinary {
				D SaveHistogram(mi,epis,graphType,graphOrder,base64Stream)
			}
		}
		K base64Stream
	}
    Q
  
 
Save  
    D Trace^MI.MIF000(mi,epis_"^"_result,"Test")
  	I $l(epis),$l(result) {
	  	S QC=""
  		D Save^MI.MIF000(mi,epis,result,date,time,QC)
  	}
  	S (sample,epis,surname,rec,res,result,date,time,QC,resultType)=""
  	Q
  
SendACK(mi,mshStr,ackType="AA",errCode=0,errDesc="Message accepted") //send 'ack' to instrument 
    N (mi,mshStr,ackType,errCode,errDesc)
    S mi=$g(mi),mshStr=$g(mshStr),ackType=$g(ackType),errCode=$g(errCode),errDesc=$g(errDesc)
    S ret=1,sb=$C(11),eb=$C(28),cr=$C(13)
    Q:('$l(mi))||('$l(mshStr)) ret
    S mshStr=$TR(mshStr,cr,"")
    //Q:$l(mshStr,"|")<21 ret
    S ret=0
    S msgRoutine=$P($P(mshStr,"|",9),"^",2)
    S seq=$P(mshStr,"|",10)
    S mshStr=$P(mshStr,"|",1,2)_"|LIS|||"_$P(mshStr,"|",6,$l(mshStr))
  
    S $P(mshStr,"|",9)="ACK^"_msgRoutine
    W sb,mshStr,cr,"MSA|"_ackType_"|"_seq_"|"_errDesc_"|||"_errCode_"|",cr,eb,cr,*-3
    D Trace^MI.MIF000(mi,$S(ackType="AA":"ACK",ackType="AE":"Error",ackType="AR":"NAK",1:"NULL"),"H-->M")
    Q ret
SendQACK(mi,mshStr,ackType="AA",errCode=0,errDesc="Message accepted") //send 'ack' to instrument 
    N (mi,mshStr,ackType,errCode,errDesc)
    S mi=$g(mi),mshStr=$g(mshStr),ackType=$g(ackType),errCode=$g(errCode),errDesc=$g(errDesc)
    S ret=1,sb=$C(11),eb=$C(28),cr=$C(13)
    Q:('$l(mi))||('$l(mshStr)) ret
    S mshStr=$TR(mshStr,cr,"")
    //Q:$l(mshStr,"|")<21 ret
    S ret=0
    S msgRoutine=$P($P(mshStr,"|",9),"^",2)
    S seq=$P(mshStr,"|",10)
    S mshStr=$P(mshStr,"|",1,2)_"|LIS|||"_$P(mshStr,"|",6,$l(mshStr))
  
    S $P(mshStr,"|",9)="QCK^"_msgRoutine
    s QACK=sb_mshStr_cr_"MSA|"_ackType_"|"_seq_"|"_errDesc_"|||"_errCode_"|"_cr_"ERR|0|"_cr_"QAK|SR|OK|"_cr_eb_cr
    W sb,mshStr,cr,"MSA|"_ackType_"|"_seq_"|"_errDesc_"|||"_errCode_"|",cr,"ERR|0|",cr,"QAK|SR|OK|",cr,eb,cr,*-3
    D Trace^MI.MIF000(mi,QACK,"H-->M")
    Q ret  
ErrHandler
	D Trace^MI.MIF000(mi,$TR($ZERROR,"^","--")_".错误代码:"_$ECODE,"Runtime Error")
    Q 
  
    //保存直方图
SaveHistogram(mi,epis,graphType,graphOrder,base64Stream)
    N (mi,epis,graphType,graphOrder,base64Stream)
    S mi=$G(mi),epis=$G(epis),graphType=$G(graphType),graphOrder=$G(graphOrder),base64Stream=$G(base64Stream)
    Q:'$L(mi) 1
    Q:'$L(epis) 1
    Q:'$L(base64Stream) 1
    Q:'$L(graphType) 1
    S:'$L(graphOrder) graphOrder=0
    S glBinaryStream=##Class(%GlobalBinaryStream).%New()
    While 'base64Stream.AtEnd {
	    S base64=base64Stream.Read()
	    S binary=$system.Encryption.Base64Decode(base64)
	    D glBinaryStream.Write(binary)
	}

	S charStream=##Class(%GlobalCharacterStream).%New()
	S maxDot=0
	While 'glBinaryStream.AtEnd {
	    S char=glBinaryStream.Read(1)
	    S ascValue=$ASCII(char)_","
	    S maxDot=maxDot+1
	    D charStream.Write(ascValue)
	}
	D charStream.Write("0")
	K glBinaryStream
	S title=graphType
	S foreColor=12
	S:graphType="PLT" foreColor=10
	S top=10
	S left=10
	S width=300
	S height=150
	S (label,maxValue,memo)=""
    S error=##Class(MI.Instrument).SaveHistogram(charStream,mi,epis,graphType,graphOrder,label,title,maxDot,maxValue,memo,foreColor,top,left,width,height)
    I $L(error) {
		D Trace^MI.MIF000(mi,$TR(error,"^","--"),"Graph Error")
	}

		D Trace^MI.MIF000(mi,"条码扫描错误,机器未识别","Order")
		Q 1

		D Trace^MI.MIF000(mi,"在检验系统未找到条码:"_barcode_"对应的标本信息,确定条码正确或标本已经接收?","Order")
		Q 1

		S tsCode=$O(^TEPI(barcode,1,tsCode))

		Q:'$D(^TMIF(mi,0,tsCode))
		S tsDesc=$P($G(^TTAB("TS",tsCode)),"\",1)

	.s fnum="0"_fnum

{
	s snum=$g(snum)
	s tnum=0
	s numlong=$l(snum)
	f i=1:1:$L(snum) d
	.s numi=$e(snum,i)

	i (numi="D")||(numi="d") s numi=13

	q numi
}

/// w ##class(YHR.Problem.ScaleConversion).scaleconversion3()

{

	s tnum=""
	f i=1:1:10 q:(snum=0)||'$l(snum)  d
	.s pnum1=snum
	.s snum=$p(snum/scale,".",1)

	.s pnum2=$$Sixteen(pnum2)
	.s tnum=pnum2_tnum
	i scale=2 d

	q tnum

	q pnum2
}

MIFCLISalphaBI(mi)
	///仪器名称:
	///仪器型号:
	///仪器厂商:
	///通信协议:	HL7
	///交互方式:	双 向
	///编 写 者:	liubin
	///编写日期:	2018-08-14
	///接    口:	TCP/IP,RS232
	///备    注:	仪器配置:选择HL7协议,散点图选择:位图   
    S mi=$G(mi) 
    Q:'$L(mi)
  
	s par11=$li(^dbo.BTMIMachineParameterD(mi),12) 		//项目分隔符
	s par10=$li(^dbo.BTMIMachineParameterD(mi),13) 	//结果分隔符
	s AntDeli=$li(^dbo.BTMIMachineParameterD(mi),14) 		//抗生素分隔符
	s SenDeli=$li(^dbo.BTMIMachineParameterD(mi),15) 		//药敏结果分隔符
	s par4="|TCP|"_$li(^dbo.BTMIMachineParameterD(mi),17)  	//端口号
    S stx=$C(2),etx=$C(3),ack=$C(6),enq=$C(5),eot=$C(4),etb=$C(23)
    S sb=$C(11),eb=$C(28),cr=$C(13)  
    S lf=$C(10),nak=$C(21),(AllRecord,result,epis)="",kk=1
  
    S (sample,epis,surname,rec,res,result,date,time,QC,errFlag,resultType)=""
    Q:$$StartUTF8^MI.MIF000(mi)
    s par4="|TCP|"_$li(^dbo.BTMIMachineParameterD(mi),10)  	//端口号
	s IP=$li(^dbo.BTMIMachineParameterD(mi),9)
	s Port=$li(^dbo.BTMIMachineParameterD(mi),10)
	s LinkTime=0
	s Device="|TCP|"_Port
	F {
		try{
			;d Trace^MI.MIF000(mi,$ZA,"ZA")
			//未启动客户端/服务端0，客户端断开连接标志20484，无客户端连接20480，有客户端连接28674，连接并建立数据传输28672
			//lis断开
			i $ZA="0" {
				h 2
				open Device:(IP:Port:/IOTABLE="GB18030"):20
				use Device
				d Trace^MI.MIF000(mi,"lis断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"lisReConnect")
				//throw ##class(%Exception.General).%New("ConnectError","D",,"重新连接失败")
				}
			//仪器断开，或者无数据1min，重启tcp
			;i (($ZA="20484")||(LinkTime=6)) {
			i (($ZA="16388")) {
				close par4
				d Trace^MI.MIF000(mi,"仪器断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"blockConnect")
				h 2
				open Device:(IP:Port:/IOTABLE="GB18030"):20
				use Device
				d Trace^MI.MIF000(mi,"重新连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"reConnect")
				s LinkTime=0
				continue
				}
			//连接无数据传输，main每执行6次，即1min，重启tcp
			i $ZA="28674" s LinkTime=$i(LinkTime)
			//正常传输了，就清空计数
			i $ZA="28672" s LinkTime=0
			d Main
	 	}catch ex{
		 s exception="错误:"_ex.Location_"^"_ex.Data_ex.Name
		 d Trace^MI.MIF000(mi,exception,"读取或写入异常，重启服务端")
	 	}
		Q:$$Stop^MI.MIF000(mi)
	}
    C par4
    Q
    /*
	F {
		D Main
		Q:$$Stop^MI.MIF000(mi)
	}
    C par4
    Q
    */

   SET $ZTRAP="ErrHandler",iError=0
	s record=$$Read^MI.MIF000(mi,sb,eb) Q:'$l(record)
    Q:record=cr
    D Trace^MI.MIF000(mi,record,"H<--M")
    S (sample,epis,surname,rec,res,result,date,time,QC,errFlag,resultType,msgType,barcode,mshStr)=""
    S (LISMSHr,msgID,labno)=""
    F i=1:1:$L(record,cr) D  
    .S subRecord=$P(record,cr,i)
    .D Trace^MI.MIF000(mi,subRecord,"H<--M")
	.S segmentType=$P(subRecord,"|",1)
	.I segmentType="MSH" d //信息头信息
	..//MSH|^~\&|||||20180814165535||QRY^Q02|2|P|2.3.1||||||ASCII|
	..S mshStr=subRecord
	..S msgType=$P($P(subRecord,"|",9),"^",1)  //ORU,ACK,QRY,QCK,DSR
    ..I msgType="ORU" D SendACK(mi,subRecord,"AA",0,"Message accepted")
    .I segmentType="OBR" d 
	..S epis=$P(subRecord,"|",3) 
    ..S:'$l(epis) epis=+$P(subRecord,"|",4)
    ..D Trace^MI.MIF000(mi,subRecord,"subRecord")
    ..D Trace^MI.MIF000(mi,epis,"epis")
    .I segmentType="OBX" d
	..S dataTye=$P(subRecord,"|",3)  //NM (numeric) 表示数字值，用于定量项目,
    ..S channel=$p( $P(subRecord,"|",4),"^",1)
    ..S value=$P(subRecord,"|",6)
    ..d Conversion
    ..i value["^" S value=$P(value,"^",3)
	..i $l(channel),$l(value) S result=result_channel_$c(92)_value_$c(44)
	.I segmentType="QRD" d
	..s barcode=$P(subRecord,"|",9)
	.I segmentType="QRF" d	//就诊信息
	..D SendQACK(mi,mshStr,"AA",0,"Message accepted")
	..;h 0.5
	..d SendDSR(mi,record,"AA",0,"Message accepted",mshStr,barcode)
	I $l(epis),$l(result) d
    .D Trace^MI.MIF000(mi,epis_"$$"_result,"Result")
  	.D Save^MI.MIF000(mi,epis,result,date,time,QC)
  	.S (sample,epis,surname,rec,res,result,date,time,QC,resultType)=""
  	Q

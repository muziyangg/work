      /// Creator：      Huhm
     /// CreatDate：    20150528
     /// Description:   从通讯端口读取数据
     /// Table：        无
     /// Input：        仪器ID,开始符,结束符,是否保存所有数据,是否显示控制字符,超时时间值
     /// Output：       无
     /// Return：       0
     /// Others：      
Read(mi,StartChar,EndChar,AllData,ControlChar,TimeOut) n (mi,StartChar,EndChar,AllData,ControlChar,TimeOut)
    s StartChar=$g(StartChar),EndChar=$g(EndChar),AllData=$g(AllData),ControlChar=$g(ControlChar),TimeOut=$g(TimeOut)
    s (Line,Flag)=0
    i '$l(mi) q ""
    s:'TimeOut TimeOut=10 
    s:'$l(StartChar) Flag=1
    SET $ZTRAP="ErrRead",$ECODE=""  //捕获错误开始 
    //循环读取一直读到字符包含结束字符
    s Result="" f  r *x:TimeOut q:(","_EndChar_",")[(","_$c(x)_",")  q:x=-1  i x d
    .d ACKE
    .//记录数据到globle
    .i $l($g(mi)),AllData="Y" d
    ..i 'Line s Line=$o(^TMPMACHDATA(mi,99,""),-1)+1
    ..s ^TMPMACHDATA(mi,99,Line)=$g(^TMPMACHDATA(mi,99,Line))_$c(x)
    .//读的字符没包含开始字符就退出，不累积到要的数据里
    .i 'Flag,(","_StartChar_",")[(","_$c(x)_",") s Flag=1 q
    .//是否要0-32的不可见字符
    .i ControlChar="Y",x<32 q
    .//累积数据
    .i Flag s Result=Result_$c(x)
    q Result 
  
ACK	    ; send 'ack' to instrument

ErrRead
    s Error=$ze
    s Max=200
    s Date=$zd(+$h,3),Time=$zt($p($h,",",2))
    s Num=$o(^TMPMACHDATA(+mi,98,""),-1)+1,^TMPMACHDATA(+mi,98,Num)=Date_" "_Time_" : Read : "_Error
    f j=Num-Max:-1 q:'$d(^TMPMACHDATA(+mi,98,j))  k ^TMPMACHDATA(+mi,98,j)
    h 3     
    q ""

MIFSnibeLisNew(mi)
    ///仪器名称:    新产业流水线
    ///仪器型号:    SnibeLis
    ///仪器厂商:    新产业
    ///通信协议:    HL7
    ///交互方式:    双 向
    ///接    口:    TCP/IP,RS232    
    ///备    注:    仪器配置:选择HL7协议,散点图选择:位图   
    S mi=$G(mi)
    S ^TMPMACH10(mi,104)=$H 
    Q:'$L(mi)
  
    s par11=$li(^dbo.BTMIMachineParameterD(mi),12)         //项目分隔符
    s par10=$li(^dbo.BTMIMachineParameterD(mi),13)     //结果分隔符
    s AntDeli=$li(^dbo.BTMIMachineParameterD(mi),14)         //抗生素分隔符
    s SenDeli=$li(^dbo.BTMIMachineParameterD(mi),15)         //药敏结果分隔符
    s par4="|TCP|"_$li(^dbo.BTMIMachineParameterD(mi),17)      //端口号

    S mi1=mi
    S stx=$C(2),etx=$C(3),ack=$C(6),enq=$C(5),eot=$C(4),etb=$C(23)
    S sb=$C(11),eb=$C(28),cr=$C(13)  
    S lf=$C(10),nak=$C(21),(AllRecord,result,epis)="",kk=1
    S $ZT="ErrHandler",$EC=""
    S (sample,epis,surname,rec,res,result,date,time,QC,errFlag,resultType)=""
    S ^TMPMACH10(mi,102)=$H
    
    Q:$$Start^MI.MIF000(mi)
	s par4="|TCP|"_$li(^dbo.BTMIMachineParameterD(mi),10)  	//端口号
	s IP=$li(^dbo.BTMIMachineParameterD(mi),9)
	s Port=$li(^dbo.BTMIMachineParameterD(mi),10)
	s LinkTime=0
	s Device="|TCP|"_Port
	F {
		try{
			;d Trace^MI.MIF000(mi,$ZA,"ZA")
			i (($ZA="28674")) {
				try{d WAck
				}catch ex{
					 s exception="错误:"_ex.Location_"^"_ex.Data_ex.Name
					 d Trace^MI.MIF000(mi,exception,"读取或写入异常，重启服务端")
					 
					 close par4
					d Trace^MI.MIF000(mi,"仪器断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"blockConnect")
					;h 2
					open Device:(IP:Port:/IOTABLE="GB18030"):20
					use Device
					d Trace^MI.MIF000(mi,"重新连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"reConnect")
					s LinkTime=0
					continue
					}
			}
			
			//未启动客户端/服务端0，客户端断开连接标志20484，无客户端连接20480，有客户端连接28674，连接并建立数据传输28672
			//lis断开
			i $ZA="0" {
				;h 2
				open Device:(IP:Port:/IOTABLE="GB18030"):20
				use Device
				d Trace^MI.MIF000(mi,"lis断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"lisReConnect")
				//throw ##class(%Exception.General).%New("ConnectError","D",,"重新连接失败")
				}
			//仪器断开，或者无数据1min，重启tcp
			;i (($ZA="20484")||(LinkTime=6)) {
			i 0,(($ZA="20484")) {
				close par4
				d Trace^MI.MIF000(mi,"仪器断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"blockConnect")
				;h 2
				open Device:(IP:Port:/IOTABLE="GB18030"):20
				use Device
				d Trace^MI.MIF000(mi,"重新连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"reConnect")
				s LinkTime=0
				continue
				}
			//连接无数据传输，main每执行6次，即1min，重启tcp
			i $ZA="28674" s LinkTime=$i(LinkTime)
			//正常传输了，就清空计数
			i $ZA="28672" d
			.s LinkTime=0
			.
			d Main
	 	}catch ex{
		 s exception="错误:"_ex.Location_"^"_ex.Data_ex.Name
		 d Trace^MI.MIF000(mi,exception,"读取或写入异常，重启服务端")
		 
		 close par4
		d Trace^MI.MIF000(mi,"仪器断开连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"blockConnect")
		;h 2
		open Device:(IP:Port:/IOTABLE="GB18030"):20
		use Device
		d Trace^MI.MIF000(mi,"重新连接"_$zd($p($h,",",1),3)_" "_$zt($p($h,",",2)),"reConnect")
		s LinkTime=0
		continue
				
	 	}
		Q:$$Stop^MI.MIF000(mi)
	}
    C par4
    q
    
WAck
   s ack=$c(6)   ;_$c(94)_$c(82)_$c(48)_$c(49)
   w ack,*-3
   q
    /*
    Q:$$Start^MI.MIF000(mi)
    F {
        D Main
        Q:$$Stop^MI.MIF000(mi)
    }
    C par4
    Q*/
ErrHandler
    h 10
    d Trace^MI.MIF000(mi,$TR($ZERROR,"^","--")_".错误代码:"_$ECODE,"Runtime Error")
    Q
Main
    SET $ZTRAP="ErrHandler",$EC=""
    R *R:10
    Q:'$TEST    //如果超时则退出
    Q:$c(R)'=sb   //以sb开头
    S (sample,epis,surname,rec,res,result,date,time,QC,errFlag,resultType,msgType,barcode,mshStr,Posi,qrdStr,MachineFlag,mid)=""
    S startTime=$P($H,",",2)
    K PLIST
    S index=1
    F {
        S curTime=$P($H,",",2)
        Q:curTime>(startTime+300)  //5分钟内没有处理完则退出循环
        Q:$$Stop^MI.MIF000(mi)
        R *R:10
        Q:'$TEST    //如果超时则退出
        I $L($G(PLIST(index)))>=32700 {  //32767
            S index=+$O(PLIST(""),-1)+1
            S PLIST(index)=""
        }
        I $C(R)'=eb,$C(R)'=cr {
            S PLIST(index)=$G(PLIST(index))_$C(R)
        }
        I $C(R)=cr {      //解析一条记录
            D Record
            K PLIST
            S index=1    
        }    
        Q:$C(R)=eb       //循环读取,当为eb的时候则退出
    }
    D Trace^MI.MIF000(mi,result,"result")
    D Save
    Q
    
Record
    S index=""
    S segmentType="",dataTye="",graphType="",base64Stream=""
    S isImage=0,isBinary=0
    F {
        S index=$O(PLIST(index))
        Q:'$L(index)
        S record=$TR($G(PLIST(index)),cr)
        Continue:'$L(record)
        D Trace^MI.MIF000(mi,record,"H<--M")
        I index=1 {
            S segmentType=$P(record,"|",1)
            I segmentType="MSH" {    //信息头信息
                S mshStr=record
                //消息类型
                S msgType=$P($P(record,"|",9),"^",1)  //ORU,ACK,QRY,QCK,DSR    
                I msgType="OUL" {
                    D SendACK(mi,record,"AA",0,"Message accepted")
                }
                I msgType="ORM" {
                }
                S resultType=$P(record,"|",11)
                S dateStr=$P(record,"|",10)
                I dateStr?14N {
                   S date=$ZDH($E(dateStr,1,4)_"-"_$E(dateStr,5,6)_"-"_$E(dateStr,7,8),3)
                   S time=$ZTH($E(dateStr,9,10)_":"_$E(dateStr,11,12)_":"_$E(dateStr,13,14),1) 
                }
            }
            
            I segmentType="PID" {    //患者信息
                    
            }
            
            I segmentType="PVI" {    //就诊信息
                    
            }
            
            I segmentType="SPM" {   //标本信息
            	s ResultFlag=$P(record,"|",12)		//P:病人样本,Q:质控品
            	S epis=$P($P(record,"|",3),"^",1)
            	s QCepis=$P($P(record,"|",3),"^",3)
            	i ResultFlag="Q" d
            	.s epis=QCepis
            }
            
            I segmentType="ORC" {   //日期信息
               S dataTye=$P(record,"|",3)  //NM (numeric) 表示数字值，用于定量项目,
                                           //ST (string) 表示字符串值，用于定性项目
            }
            
            I segmentType="OBX" {
               S channel=$P(record,"|",4)
               S value=$P(record,"|",6)
               s MachID=$P(record,"|",19)
               s result=channel_par10_value_par11
            }
            
            //根据条码号获取LIS标本
            I segmentType="QPD" {
                S barcode=$P($P(record,"|",4),"^",1)  //QPD| TSREQ ||20251122553^1556^N0002^2|2
                ;s barcode=$P(record,"^",1)
                s qrdStr=$P(record,"|",4)
                s RackNo=$p(qrdStr,"^",3)_"-"_$p(qrdStr,"^",4)
 				s positioninfo="新产业-"_RackNo
				try{j ##class(MI.Special.Common).SaveRackNo(barcode, positioninfo, mi, "上机")}catch{}		//YHR 202050330 仪器上机上传架子号增加上机标志
				D Trace^MI.MIF000(mi,barcode_"$$"_RackNo,"barcode")
                d SendORR(mi,record,"AA",0,"Message accepted",mshStr,barcode)
            }
                
        }
        
        I dataTye="ED",index>1,$L(base64Stream) {
            D base64Stream.Write($P(record,"|",1))    
        }
    }
    Q
  
 
Save  ; file result if exist 
  i $l(epis),$l(result) d 
  .d Trace^MI.MIF000(mi,epis_"$$"_result,"结果")
  .d Save^MI.MIF000(mi,epis,result,date,time,QC)
  q
